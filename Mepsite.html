üìú MEPLua Scripting Language

MEPLua is a Lua-based scripting language designed for game development in MEP Sandbox Engine. It provides a familiar API for Roblox developers while adding unique features.

Script Lifecycle

on_start()

Called once when Play is pressed. Use for initialization:

function on_start()
    print("Game started!")
    script.Parent.Color = Color3.new(1, 0, 0)
end

on_update(dt)

Called every frame (~60 FPS). dt = delta time in seconds:

local rotation = 0

function on_update(dt)
    rotation = rotation + (90 * dt)  -- 90¬∞ per second
    script.Parent.Rotation = Vector3.new(0, rotation, 0)
end

üí° Code Examples

Kill Brick

Instant-death obstacle:

local killBrick = script.Parent

function on_start()
    killBrick.Color = Color3.fromRGB(255, 0, 0)
    killBrick.Touched:Connect(onTouched)
end

function onTouched(hit)
    local humanoid = hit.Parent:FindFirstChild("Humanoid")
    if humanoid then
        humanoid.Health = 0
        print(hit.Parent.Name .. " eliminated!")
    end
end

Tycoon Dropper

Spawn parts at intervals:

local dropper = script.Parent
local dropRate = 2  -- seconds
local timer = 0

function on_update(dt)
    timer = timer + dt
    
    if timer >= dropRate then
        timer = 0
        spawnPart()
    end
end

function spawnPart()
    local part = Instance.new("Part")
    part.Name = "DroppedPart"
    part.Size = Vector3.new(1, 1, 1)
    part.Color = Color3.fromRGB(255, 215, 0)
    part.Position = dropper.Position + Vector3.new(0, -2, 0)
    part.Parent = workspace
    print("Dropped a part!")
end

Damage Zone

Continuous damage over time:

local damageZone = script.Parent
local damagePerSecond = 25
local playersInZone = {}

function on_start()
    damageZone.Touched:Connect(function(hit)
        local humanoid = hit.Parent:FindFirstChild("Humanoid")
        if humanoid then
            playersInZone[humanoid] = true
        end
    end)
end

function on_update(dt)
    for humanoid, _ in pairs(playersInZone) do
        if humanoid.Health > 0 then
            humanoid.Health = humanoid.Health - (damagePerSecond * dt)
        else
            playersInZone[humanoid] = nil
        end
    end
end

Teleporter

Transport between locations:

local entrance = script.Parent
local exitPosition = Vector3.new(50, 5, 50)

function on_start()
    entrance.Color = Color3.fromRGB(255, 200, 0)
    entrance.Touched:Connect(onTouched)
end

function onTouched(hit)
    if hit.Parent:FindFirstChild("HumanoidRootPart") then
        local rootPart = hit.Parent.HumanoidRootPart
        rootPart.Position = exitPosition
        print("Teleported " .. hit.Parent.Name)
    end
end

Healing Zone

Restore health:

local healZone = script.Parent
local healPerSecond = 20

function on_start()
    healZone.Color = Color3.fromRGB(0, 255, 100)
    healZone.Touched:Connect(onTouched)
end

function onTouched(hit)
    local humanoid = hit.Parent:FindFirstChild("Humanoid")
    
    if humanoid and humanoid.Health < humanoid.MaxHealth then
        humanoid.Health = math.min(
            humanoid.Health + healPerSecond,
            humanoid.MaxHealth
        )
    end
end

Spinning Part

Continuous rotation:

local part = script.Parent
local rotationSpeed = 90  -- degrees per second
local rotation = 0

function on_update(dt)
    rotation = rotation + (rotationSpeed * dt)
    part.Rotation = Vector3.new(0, rotation, 0)
end

Color Cycle

Rainbow effect:

local part = script.Parent
local hue = 0

function on_update(dt)
    hue = hue + (0.2 * dt)  -- Cycle through colors
    if hue > 1 then hue = 0 end
    
    part.Color = Color3.fromHSV(hue, 1, 1)
end
