The Lifecycle

-- on_start() runs once when the game loads
function on_start()
    print("Engine Initialized")
    script.Parent.Color = Color3.new(1, 1, 1) -- Set to White
end

-- on_update(dt) runs every frame
function on_update(dt)
    -- Rotate object on Y axis
    local currentRot = script.Parent.Rotation
    script.Parent.Rotation = Vector3.new(currentRot.X, currentRot.Y + (45 * dt), currentRot.Z)
end

Interactive Events

-- Handle physics collisions
script.Parent.Touched:Connect(function(otherPart)
    print("Collision Detected with: " .. otherPart.Name)
    
    local humanoid = otherPart.Parent:FindFirstChild("Humanoid")
    if humanoid then
        humanoid.Health = 0 -- Eliminate player
    end
end)
