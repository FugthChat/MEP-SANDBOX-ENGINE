üìò Core Concepts

Every script in MEP has access to a lifecycle system and event handlers. Scripts are attached to objects and can reference their parent, children, and other scene objects.

üîÑ The Lifecycle System

-- on_start() executes once when the scene loads
function on_start()
    print("Game Initialized!")
    
    -- Access the object this script is attached to
    script.Parent.Color = Color3.new(1, 0, 0) -- Set to red
    script.Parent.Position = Vector3.new(0, 5, 0) -- Move up
    
    -- Store data in the script
    script.speed = 5
    script.timer = 0
end

-- on_update(dt) runs every frame (dt = delta time)
function on_update(dt)
    -- Smooth rotation over time
    local rotation = script.Parent.Rotation
    local newY = rotation.Y + (script.speed * dt * 10)
    script.Parent.Rotation = Vector3.new(rotation.X, newY, rotation.Z)
    
    -- Oscillating movement
    script.timer = script.timer + dt
    local yPos = math.sin(script.timer * 2) * 3
    script.Parent.Position = Vector3.new(0, yPos, 0)
end

üéÆ Player & Character Control

-- Character Movement Script
function on_start()
    script.humanoid = script.Parent:FindFirstChild("Humanoid")
    script.moveSpeed = 16
end

function on_update(dt)
    local player = game.Players.LocalPlayer
    if not player then return end
    
    local character = player.Character
    if not character then return end
    
    -- Get input (W, A, S, D)
    local moveDirection = player:GetMoveVector()
    
    -- Apply movement
    character.HumanoidRootPart.Position = 
        character.HumanoidRootPart.Position + (moveDirection * script.moveSpeed * dt)
end

üîç Scene Manipulation

-- Find and manipulate objects in the scene
function on_start()
    -- Find object by name
    local targetPart = game.Workspace:FindFirstChild("Target")
    if targetPart then
        targetPart.Color = Color3.new(1, 1, 0) -- Yellow
    end
    
    -- Get all children
    local allParts = game.Workspace:GetChildren()
    for i, part in ipairs(allParts) do
        print(part.Name)
    end
    
    -- Create new objects dynamically
    local newPart = Instance.new("Part")
    newPart.Name = "DynamicPart"
    newPart.Position = Vector3.new(0, 10, 0)
    newPart.Parent = game.Workspace
end
