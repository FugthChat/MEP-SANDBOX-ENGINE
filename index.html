MEPLua Scripting

Write clean, efficient code that runs in real-time

-- Rotate and animate an object
function on_start()
    print("Scene initialized!")
    script.Parent.Color = Color3.new(1, 0, 0)
    script.speed = 5
    script.timer = 0
end

function on_update(dt)
    -- Smooth rotation
    local rotation = script.Parent.Rotation
    local newY = rotation.Y + (script.speed * dt * 10)
    script.Parent.Rotation = Vector3.new(rotation.X, newY, rotation.Z)
    
    -- Oscillating movement
    script.timer = script.timer + dt
    local yPos = math.sin(script.timer * 2) * 3
    script.Parent.Position = Vector3.new(0, yPos, 0)
end
